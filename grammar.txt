all                     -> rounds

rounds                  -> rounds round
                        | round

round                   -> ROUND NUMBER : expressions

expressions             -> expressions, aliasable_expression
                        | aliasable_expressions

aliasable_expression    -> expression ( COUNTS_AS alias)
                        | expression TO_MAKE alias
                        | expression

expression              -> stitch_type
                        | CHAIN NUMBER
                        | NUMBER stitch_type
                        | NUMBER stitch_type INTO destination
                        | stitch_type INTO destination
                        | loop
                        | RING
                        | SKIP stitch_type
                        | SKIP NEXT stitch_type
                        | SKIP NUMBER stitch_type

alias                   -> RING
                        | simple_stitch
                        | FIRST simple_stitch

stitch_type             -> simple_stitch
                        | grouped_stitch

simple_stitch           -> SC
                        | DC
                        | TR
                        | DTR
                        | HDC
                        | SLIP_STITCH
                        | CHAIN
                        | STITCH

grouped_stitch          -> NUMBER stich INCREASE
                        | NUMBER stitch DECREASE
                        | NUMBER stitch POPCORN
                        | NUMBER stitch CLUSTER


destination             -> NEXT stitch_type
                        | LAST stitch_type
                        | FIRST stitch_type
                        | SAME stitch_type
                        | ORDINAL CHAIN FROM_HOOK
                        | RING



loop                    -> * expressions REPEAT * number_times
                        | * expressions REPEAT from * AROUND
                        | [ expressions ] number_times
                        | [ expressions ] AROUND

number_times            -> NUMBER TIMES
                        | TWICE
